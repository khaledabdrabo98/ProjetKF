54  void create(const std::vector<Mesh> &tabMeshes)
55  {
56  size_t id = 0;
57  size_t offset = 0;
58  size_t size = 0;
59  // cree et configure un vertex array object: conserve la description des attributs de sommets
60  glGenVertexArrays(1, &vao);
61  glBindVertexArray(vao);
62  for (unsigned int i = 0; i < tabMeshes.size(); i++)
63  {
64  Mesh mesh = tabMeshes.at(i);
65 
66  id = i;
67 
68  if (!mesh.vertex_buffer_size()) return;
69 
70  // creer, initialiser le buffer : positions
71  glGenBuffers(1, &tab_vertex_buffer[i]);
72  glBindBuffer(GL_ARRAY_BUFFER, tab_vertex_buffer[i]);
73 
74  // taille totale du buffer
75  size = mesh.vertex_buffer_size();
76  glBufferData(GL_ARRAY_BUFFER, size, nullptr, GL_STATIC_DRAW);
77 
78  // transfere les positions des sommets
79 
80  glBufferSubData(GL_ARRAY_BUFFER, offset, size, mesh.vertex_buffer());
81  // et configure l'attribut id, vec3 position
82  glVertexAttribPointer(id, 3, GL_FLOAT, GL_FALSE, /* stride */ 0, (const GLvoid *)offset);
83  glEnableVertexAttribArray(id);
84 
85  std::cout << "id : " << id << "\n";
86  vertex_count += mesh.vertex_count();
87  }
88 
89  // Ajout des normales
90  if (tabMeshes.at(0).normal_buffer_size()){
91  id += 1;
92  glGenBuffers(1, &vertex_normals_buffer);
93  glBindBuffer(GL_ARRAY_BUFFER, vertex_normals_buffer);
94 
95  // taille totale du buffer

103  glEnableVertexAttribArray(id);

105  }
